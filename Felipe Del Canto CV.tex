\newcommand{\setCols}[1]{			%
	\ifnum0=\i						%
		\ifdim0cm=#1				%
			\def \firstCol {r}		%
		\else						%
			\def \firstCol {p{#1}}		%
		\fi						%
	\else \ifnum1=\i				%
		\ifdim0cm=#1				%
			\def \secondCol {l}		%
		\else						%
			\def \secondCol{p{#1}}	%
		\fi						%
	\else \ifnum2=\i				%
		\ifnum0=#1				%
			\def \sep {}			%
		\else						%
			\def \sep {|}			%
		\fi						%
	\fi \fi \fi						%
	\advance\i by1					%
}
